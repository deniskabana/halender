const   defaultPinnedColumns :   Record < string ,   unknown [] >   =   {

   left :   [ gridIndexColumn . field ] ,   // Default pinned column on the left

}   as   const   satisfies   PinnedColumns ;

  

/**

 * The pinning of columns in data grid (feature design)

 * @warning  Will always pin index column first

 */

export   default function   useColumnPinning < Cols   extends   unknown [] > ( columns :   Cols )   {

   const   [ pinnedColumns ,   setPinnedColumns ]   =   useState < PinnedColumns   |   null > ( null ) ;

 

   const   handlePinnedColumnsChange   =   ( newPinnedColumns :   PinnedColumns )   =>   {

     setPinnedColumns ( ( )   =>   {

       if   ( columns . length   ===   0 )   throw new   Error ( "Incorrect input." ) ;

 

       const   newLeft   =   new   Set ( newPinnedColumns . left ) ;

       newLeft . delete ( newPinnedColumns . left [ 0 ] ) ;   // Enforce always first

       return   {   left :   [ newPinnedColumns . left [ 0 ] ,   ... newLeft ]   } ;

     } ) ;

   } ;

 

   return   {   handlePinnedColumnsChange ,   pinnedColumns   } ;

}
